### Data fetching useEffect (depends on provider and refreshTrigger)
```tsx
useEffect(() => {
  let cancelled = false;
  setIsLoading(true);
  setError(null);
  fetch(buildConsoleStateUrl(provider))
    .then((res) => { if (!res.ok) throw new Error(`Console state error: ${res.status}`); return res.json(); })
    .then((data) => {
      if (cancelled) return;
      const agentMap: Record<string, Agent> = {};
      for (const agent of data.agents ?? []) agentMap[agent.id] = agent;
      setAgents(agentMap);
      setEdges(data.edges ?? []);
      setMeshStatus(data.mesh ?? null);
      setUsageData(data.usage ?? null);
      setTotalAgentCount(data.agents?.length ?? 0);
      setIsLoading(false);
    })
    .catch((err) => {
      if (cancelled) return;
      setError(err instanceof Error ? err.message : "Failed to load console state");
      setIsLoading(false);
    });
  return () => { cancelled = true; };
}, [provider, refreshTrigger]);
```

### Fetch agent detail effect (depends on selectedAgentId and provider)
```tsx
useEffect(() => {
  if (!selectedAgentId) { setSelectedAgentDetail(null); return; }
  let cancelled = false;
  setIsDetailLoading(true);
  setDetailError(null);
  fetch(buildConsoleAgentDetailUrl(selectedAgentId, provider))
    .then((res) => { if (!res.ok) throw new Error(`Agent detail error: ${res.status}`); return res.json(); })
    .then((data) => { if (cancelled) return; setSelectedAgentDetail(data); setIsDetailLoading(false); })
    .catch((err) => { if (cancelled) return; setDetailError(err.message); setIsDetailLoading(false); });
  return () => { cancelled = true; };
}, [selectedAgentId, provider]);
```
